# 랜덤 테스트

## 어제 구현한 랜덤함수를 100만번 반복해서 각 수가 나오는 확률을 구하려고 한다.

```C#
#include <stdio.h>
#include <stdlib.h>
#include <time.h>


int sumDice[10];

// 0~9까지 랜덤으로 반환하는 함수 구현
int random() {
	int randNum = 0;

    struct timespec ts;

    // 현재 시간 가져오기
    timespec_get(&ts, TIME_UTC);

    long currentTime = ts.tv_nsec;
    randNum = currentTime / 100;
	randNum = (randNum % 10); // 0~9까지의 수로 변환

	return randNum;
}

int main() {
    for (int i = 0; i < 10000000; i++) {
        int x = random();
        sumDice[x]++;
    }

    for (int i = 0; i < 10; i++) {
        double prob = (double)sumDice[i] * 100.0 / 10000000.0;
        printf("%2d 가 나올 확률 : %.4f%%\n", i, prob);
    }
}
```
